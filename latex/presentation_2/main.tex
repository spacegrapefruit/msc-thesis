\section{Research Direction Change}

\begin{frame}{Shift in research focus}
  \begin{alertblock}{The research focus has changed.}
    \begin{itemize}
      \item \textbf{Previous:} Focus on text normalization strategies for TTS.
      \item \textbf{Current:} Training robust multi-speaker TTS models.
    \end{itemize}
  \end{alertblock}

  \vspace{1em}
  \textbf{Motivation for change:}
  \begin{itemize}
    \item Discovery of the \textit{Liepa 2} dataset.
    \item \textit{Liepa 2} provides significantly larger and higher-quality annotated data than \textit{Common Voice}.
    \item The transcriptions are pre-normalized, thus normalization is no longer a concern.
  \end{itemize}
\end{frame}

\section{Dataset \& Challenges}

\begin{frame}{Liepa 2 Dataset}
  \textbf{Dataset overview:}
  \begin{itemize}
    \item \textbf{1000 hours} of Lithuanian speech.
    \item \textbf{2621} distinct speakers.
    \item Varied recording conditions (Studio, Radio, TV, etc.).
  \end{itemize}

  \vspace{1em}
  \textbf{The challenge:}
  \begin{itemize}
    \item The ``top'' speaker has only \textbf{2.5 hours} of speech.
    \item High-quality single-speaker TTS usually requires at least \textbf{4 to 20 hours} of speech.
    \item Implication: We cannot train a high-quality model on a single speaker alone.
  \end{itemize}

  \begin{center}
    \textit{Solution: Train multi-speaker models to leverage shared linguistic features.}
  \end{center}
\end{frame}

\begin{frame}{Feasibility}
  \textbf{Computational constraints:}
  \begin{itemize}
    \item Training on the full 1000h corpus is computationally expensive and time-consuming.
    \item Limited GPU resources necessitate smaller training sets.
    \item We need strategies to select effective subsets of data.
  \end{itemize}

  \vspace{1em}
  \textbf{Research Questions:}
  \begin{enumerate}
    \item How should we select subsets of data for training multi-speaker TTS models?
    \item How does the number of speakers in the training data affect the quality of synthesized speech?
  \end{enumerate}
\end{frame}

\section{Methodology}

\begin{frame}{Methodology}
  \textbf{Models:}
  \begin{itemize}
    \item \textbf{Tacotron 2 Variants:}
          \begin{itemize}
            \item Using DCA (Dynamic Convolution Attention) and DDC (Double Decoder Consistency).
            \item Adapted for pre-trained multi-speaker (VCTK Corpus) HiFi-GAN v2 vocoder.
          \end{itemize}
    \item \textbf{VITS:} End-to-end pipeline (Glow-TTS + HiFi-GAN).
  \end{itemize}

  \vspace{1em}
  \textbf{Subset selection strategies:}
  \begin{itemize}
    \item \textbf{Top-N speakers:} Selecting only speakers with the most data.
    \item \textbf{Random sampling:} Selecting utterances uniformly at random.
    \item \textbf{Balanced sampling:} Optimizing for speaker diversity (preferring more speakers with fewer utterances each).
  \end{itemize}
\end{frame}
