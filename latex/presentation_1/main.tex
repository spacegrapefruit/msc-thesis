% --- INTRODUCTION ---
\section{Introduction}

\begin{frame}{Problem statement}
  \begin{itemize}
    \item \textbf{Text-to-Speech (TTS) systems:} Converting written text into
          human-like speech
    \item \textbf{End-to-End (E2E) neural models:} State-of-the-art approach
          achieving high quality
    \item \textbf{Challenge:} E2E models struggle with:
          \begin{itemize}
            \item Non-Standard Words (NSWs): Numbers, dates, abbreviations
            \item Low-resource languages with limited training data
            \item Morphologically complex languages like Lithuanian
          \end{itemize}
    \item \textbf{Example:} 21 (dvidesimt vieneri) metai vs. 21 (dvidesimt
          vienas) saldainis
  \end{itemize}

  \vspace{1em}
  \textbf{Research Questions:}
  \begin{enumerate}
    \item What impact does Lithuanian text normalization have on E2E TTS
          quality?
    \item What is the trade-off between normalization complexity and TTS
          performance?
  \end{enumerate}
\end{frame}

% --- RESEARCH METHODOLOGY ---
\section{Research plan \& Methodology}

\begin{frame}{Experimental design \& Evaluation}
  \textbf{Experimental design:}
  \begin{itemize}
    \item \textbf{Dataset:} Lithuanian speech--text corpora with NSWs and
          annotations: Common~Voice, Liepa~2, audiobooks (?)
    \item \textbf{Normalization levels:} Baseline (none), Rule-based (several
          levels), Neural (if data permits)
    \item \textbf{TTS models:} Popular E2E architectures (Tacotron~2,
          FastSpeech~2, VITS)
  \end{itemize}

  \vspace{1em}
  \textbf{Evaluation:}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textbf{Objective:}
      \begin{itemize}
        \item Mel cepstral distortion (MCD)
        \item Duration modeling accuracy
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{Subjective:}
      \begin{itemize}
        \item Mean Opinion Score (MOS)
        \item NSW intelligibility tests
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}
